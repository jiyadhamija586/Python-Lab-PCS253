def bfs (st,goal,graph):
    vis=set()
    queue=[st]
    path=[]
    vis.add(st)
    while queue:
        node=queue.pop(0)
        path.append(node)
        if node==goal:
            return path
        for neighbour in graph[node]:
            if neighbour not in vis:
                vis.add(neighbour)
                queue.append(neighbour)
    return path
